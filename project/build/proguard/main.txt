-include common.txt

# since this file is in the project/build/proguard directory, we need to get back up to the root
-basedirectory ../../..

-keepdirectories system

-injars target/scala_2.9.0-1/classes(!org/nlogo/lab/**)
-injars resources(!**.svn**,**)
-outjar NetLogo.jar
